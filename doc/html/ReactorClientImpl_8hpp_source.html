38  auto progressLambda = [this](reactor_progress const& i_progress, Guid const& /*sampleId*/,
39  Guid const& i_relatedId) {
40  // Try to deserialize the sample
41  bool haveData = false;
42  ProgressData sample;
43  if (i_progress.data().size() > 0) {
44  PubSubType<ProgressData> deser;
45  efr::SerializedPayload_t serialized;
46  serialized.length = i_progress.data().size();
47  serialized.data = const_cast<unsigned char*>(i_progress.data().data());
48  serialized.encapsulation = CDR_LE; // FIXME how do you know? It's in the encapsulation bytes, but
49  // we've got to dig through *OMG* documents for it :(
50  haveData = deser.deserialize(&serialized, &sample);
51  serialized.data = nullptr; // Prevent double free, as mem is owned by rawData
52  }
53  // Place the sample
54  LockGuard guard(m_mutex);
55  auto it = m_session.find(i_relatedId);
56  if (it == m_session.end()) { return; }
57  it->second.progress = i_progress.progress();
58  LT_LOG << m_participant.get() << ":" << m_service << "-client"
59  << " Session ID " << i_relatedId << " progress " << i_progress.progress() << " w/ "
60  << i_progress.data().size() << " bytes data \n";
61  if (i_progress.data().size() > 0) {
62  if (haveData) {
63  it->second.progressData.push({i_progress.progress(), std::move(sample)});
64  it->second.cvProgressData.notify_one();
65  } else {
66  LT_LOG << m_participant.get() << ":" << m_service << "-client"
67  << " Session ID " << i_relatedId << " got mangled progress data\n";
68  }
69  }
70  };
71  m_participant->subscribe<reactor_progress>(reactorProgressName(i_serviceName), progressLambda, "stateful");
72 
73  m_requestSender = m_participant->advertise<Req>(reactorRequestName(m_service), "stateful", 8);
74  m_lastId = m_requestSender.guid();
75  auto repCallback = [this](Rep const& i_reply, Guid const& /*sampleId*/, Guid const& i_relatedId) {
76  LockGuard guard(m_mutex);
77  auto it = m_session.find(i_relatedId);
78  if (it == m_session.end()) { return; }
79  LT_LOG << m_participant.get() << ":" << m_service << "-client"
80  << " Finish session ID " << i_relatedId << "\n";
81  it->second.progress = PROG_SUCCESS;
82  it->second.reply = std::move(i_reply);
83  it->second.replyReady = true;
84  guard.unlock();
85  it->second.cvReply.notify_one();
86  };
87  m_participant->subscribe<Rep>(reactorReplyName(m_service), repCallback, "stateful");
88  }
