44  void on_data_available(efd::DataReader* i_reader) override
45  {
46  if (nullptr == i_reader) { return; }
47  Req data;
48  efd::SampleInfo info;
49  while (ReturnCode_t::RETCODE_OK == i_reader->take_next_sample(&data, &info)) {
50  if (info.valid_data) {
51  Guid relatedId = toLetsTalkGuid(info.sample_identity);
52  LT_LOG << m_serviceName << ": Request " << relatedId << " received\n";
53  submitJob([this, data, relatedId]() {
54  Rep reply;
55  bool isBad = false;
56  try {
57  reply = m_providerCallback(data);
58  } catch (...) {
59  isBad = true;
60  }
61  LT_LOG << m_serviceName << ": Sending reply for " << relatedId << ", result "
62  << (isBad ? "FAILED" : "okay") << "\n";
63  m_sender.publish(reply, m_Id.increment(), relatedId, isBad);
64  });
65  }
66  }
67  }

125  void onReply(Rep const& data, Guid const& id, Guid const& relatedId)
126  {
127  Guid badId = relatedId.makeBadVersion();
128  std::unique_lock<std::mutex> guard(m_lock);
129  auto it = m_requests.find(relatedId);
130  if (it != m_requests.end()) {
131  LT_LOG << serviceName() << ": Success response received for pending request " << relatedId << "\n";
132  it->second.set_value(data);
133  m_requests.erase(it);
134  return;

136  it = m_requests.find(badId);
137  if (it != m_requests.end()) {
138  LT_LOG << serviceName() << ": Failure response received for pending request " << badId << "\n";
139  it->second.set_exception(std::make_exception_ptr(std::runtime_error("RPC failed")));
140  m_requests.erase(it);
141  return;
142  }
143  LT_LOG << serviceName() << ": Reply for " << relatedId << " (aka " << badId << ") ignored.\n";
144  }
