49  void on_data_available(efd::DataReader* i_reader) override
50  {
51  if (nullptr == i_reader) { return; }
52  Req data;
53  efd::SampleInfo info;
54  while (ReturnCode_t::RETCODE_OK == i_reader->take_next_sample(&data, &info)) {
55  if (info.valid_data) {
56  Guid relatedId = toLetsTalkGuid(info.sample_identity);
57  LT_LOG << m_serviceName << ": Request " << relatedId << " received\n";
58  submitJob([this, data, relatedId]() {
59  Rep reply;
60  bool isBad = false;
61  try {
62  reply = m_providerCallback(data);
63  } catch (...) {
64  isBad = true;
65  }
66  LT_LOG << m_serviceName << ": Sending reply for " << relatedId << ", result "
67  << (isBad ? "FAILED" : "okay") << "\n";
68  m_sender.publish(reply, m_Id.increment(), relatedId, isBad);
69  });
70  }
71  }
72  }
