90  std::pair<Guid, Req> getPendingSessionData(std::chrono::nanoseconds i_wait)
91  {
92  LockGuard guard(m_requestMutex);
93  if (m_arePending.wait_for(guard, i_wait, [this]() { return !m_pending.empty(); })) {
94  auto sessionData = std::move(m_pending.front());
95  m_pending.pop_front();
96  guard.unlock();
97  LT_LOG << m_participant.get() << ":" << m_service << "-server"

99  return sessionData;
100  }
101  return {Guid::UNKNOWN(), Req()};

120  void progress(Guid const& i_id, int i_progress, std::vector<unsigned char>& i_payload)
121  {
122  {
123  LockGuard guard(m_sessionMutex);
124  auto it = m_session.find(i_id);
125  if (it != m_session.end()) {
126  if (i_progress >= PROG_SUCCESS) {
127  it->second = SUCCEEDING;
128  } else if (i_progress > PROG_START) {
129  it->second = RUNNING;
130  } else if (i_progress == PROG_START) {
131  it->second = STARTING;
132  } else {
133  m_session.erase(it);
134  }
135  } else if (i_progress == PROG_START) {
136  m_session[i_id] = STARTING;
137  }
138  }
139  LT_LOG << m_participant.get() << ":" << m_service << "-server"
140  << " Session ID " << i_id << " progress " << i_progress << " w/ " << i_payload.size()
141  << " bytes data\n";
142  reactor_progress message;
143  message.progress(i_progress);
144  message.data().swap(i_payload);
145  m_progressSender.publish(message, m_progressSender.guid(), i_id);
146  }

150  {

152  auto it = m_session.find(i_id);
153  if (it == m_session.end()) { return false; }
154  switch (it->second) {
