140 {
141  if (!m_isGetKeyDefined) { return false; }
142 
143  T* p_type = static_cast<T*>(data);
144 
145  // Object that manages the raw buffer.
146  eprosima::fastcdr::FastBuffer fastbuffer(reinterpret_cast<char*>(m_keyBuffer), T::getKeyMaxCdrSerializedSize());
147 
148  // Object that serializes the data.
149  eprosima::fastcdr::Cdr ser(fastbuffer, ::eprosima::fastcdr::Cdr::BIG_ENDIANNESS);
150  p_type->serializeKey(ser);
151  if (force_md5 || T::getKeyMaxCdrSerializedSize() > 16) {
152  m_md5.init();
153  m_md5.update(m_keyBuffer, static_cast<unsigned int>(ser.getSerializedDataLength()));
154  m_md5.finalize();
155  for (uint8_t i = 0; i < 16; ++i) { handle->value[i] = m_md5.digest[i]; }
156  } else {
157  for (uint8_t i = 0; i < 16; ++i) { handle->value[i] = m_keyBuffer[i]; }
158  }
159  return true;
